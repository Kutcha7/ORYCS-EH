#General configuration parameters for model EcoHyD

Site: EH			[-]		name of the site
Latitude: 19.25			[°]		latitude of the site
simYears: 3			[years]		total simulation time
xsize: 30			[-]		x length of the grid (number of x-cells)
ysize: 30			[-]		y length of the grid (number of y-cells)
cellsize: 5			[m]		size of a single cell (length of a side of one cell)
vegTimeStep: 14			[days]		number of days after which vegetation is updated
